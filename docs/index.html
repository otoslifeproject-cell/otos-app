async function submitCommand() {

  const status = document.getElementById("status");

  status.textContent = "Submitting Parent command…";


  let payload;

  try {

    payload = JSON.parse(document.getElementById("payload").value);

  } catch (e) {

    status.innerHTML = '<span class="status-error">Invalid JSON payload</span>';

    return;

  }


  try {

    const res = await fetch("https://otos-app.vercel.app/api/parent", {

      method: "POST",

      headers: { "Content-Type": "application/json" },

      body: JSON.stringify(payload)

    });


    const text = await res.text();


    let data;

    try {

      data = JSON.parse(text);

    } catch {

      status.innerHTML =

        '<span class="status-error">Server returned non-JSON:</span>\n\n' +

        text.slice(0, 500);

      return;

    }


    if (!res.ok) {

      status.innerHTML =

        '<span class="status-error">Error</span>\n\n' +

        JSON.stringify(data, null, 2);

      return;

    }


    status.innerHTML =

      '<span class="status-ok">Success</span>\n\n' +

      JSON.stringify(data, null, 2);


  } catch (err) {

    status.innerHTML =

      '<span class="status-error">Network error</span>\n\n' +

      err.message;

  }

}

</script>
